Basic Database Related Queries:

1. create database db_name;
            or
   create database if not exists db_name; //Best Practices

2. drop database db_name;
       or
   drop database if exists db_name; //Best Practices

3. show databases; //Display all databases.

4. show tables; //Display all tables.

5. CREATE table_name( //Creating table
    id INT primary key,
    name varchar(50),
    age int not null
);

6. USE db_name; //Using a particular db.

7. SELECT * from table_name; //View all columns.

8. INSERT into table_name VALUES (arg1,arg2,arg3); //Inserting values to table.
